```

services:
 db:
   container_name: pg12
   image: postgres:12
   environment:
     POSTGRES_USER: nikulinm
     POSTGRES_PASSWORD: 123456
     POSTGRES_DB: netology_db
   ports:
     - "5432:5432"

     - database_volume:/home/database/
```
